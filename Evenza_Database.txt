CREATE TABLE universities (
    university_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(150) NOT NULL,
    short_name VARCHAR(50),
    email VARCHAR(100),
    phone VARCHAR(20),
    address TEXT,
    logo VARCHAR(255),
    status ENUM('active','inactive') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE colleges (
    college_id INT AUTO_INCREMENT PRIMARY KEY,
    university_id INT NOT NULL,
    name VARCHAR(150) NOT NULL,
    slug VARCHAR(150) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    address TEXT,
    logo VARCHAR(255),
    status ENUM('active','inactive') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (university_id) REFERENCES universities(university_id) ON DELETE CASCADE
);
CREATE TABLE students (
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    college_id INT NOT NULL,
    enrollment_no VARCHAR(50) UNIQUE,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    gender ENUM('male','female','other'),
    semester VARCHAR(20),

    id_card_image VARCHAR(255) NOT NULL,

    profile_photo VARCHAR(255),

    status ENUM('pending','verified','rejected','active','inactive','blocked') DEFAULT 'pending',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (college_id)
    REFERENCES colleges(college_id)
    ON DELETE CASCADE
);

CREATE TABLE categories (
    category_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    status ENUM('active','inactive') DEFAULT 'active'
);

CREATE TABLE events (
    event_id INT AUTO_INCREMENT PRIMARY KEY,
    college_id INT NOT NULL,
    category_id INT,
    title VARCHAR(150) NOT NULL,
    description TEXT,
    event_type ENUM('solo','team') NOT NULL,
    min_team_size INT DEFAULT 1,
    max_team_size INT DEFAULT 1,
    fee_type ENUM('per_person','per_team') NOT NULL,
    registration_fee DECIMAL(10,2) DEFAULT 0.00,
    registration_deadline DATETIME,
    event_date DATE,
    start_time TIME,
    end_time TIME,
    venue VARCHAR(200),
    dress_code VARCHAR(255),
    status ENUM('draft','published','completed','cancelled') DEFAULT 'draft',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (college_id) REFERENCES colleges(college_id) ON DELETE CASCADE,
    FOREIGN KEY (category_id) REFERENCES categories(category_id) ON DELETE SET NULL
);

CREATE TABLE event_rules (
    rule_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    rule TEXT NOT NULL,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

CREATE TABLE event_rounds (
    round_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    round_name VARCHAR(100),
    description TEXT,
    duration_minutes INT,
    sequence_no INT,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

CREATE TABLE event_prizes (
    prize_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    position ENUM('winner','runner_up','second_runner_up'),
    prize VARCHAR(255),
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

CREATE TABLE event_contacts (
    contact_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    person_name VARCHAR(100),
    phone VARCHAR(20),
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

CREATE TABLE teams (
    team_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    leader_id INT NOT NULL,
    team_name VARCHAR(100) NOT NULL,
    team_code VARCHAR(30) UNIQUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE,
    FOREIGN KEY (leader_id) REFERENCES students(student_id) ON DELETE CASCADE
);

CREATE TABLE team_members (
    team_member_id INT AUTO_INCREMENT PRIMARY KEY,
    team_id INT NOT NULL,
    student_id INT NOT NULL,
    UNIQUE (team_id, student_id),
    FOREIGN KEY (team_id) REFERENCES teams(team_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE
);

CREATE TABLE registrations (
    registration_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    student_id INT NULL,
    team_id INT NULL,
    registration_type ENUM('solo','team'),
    status ENUM('pending','approved','cancelled') DEFAULT 'pending',
    registered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (team_id) REFERENCES teams(team_id) ON DELETE CASCADE
);

CREATE TABLE payments (
    payment_id INT AUTO_INCREMENT PRIMARY KEY,
    registration_id INT NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    payment_method ENUM('cash','upi','card','bank_transfer'),
    transaction_id VARCHAR(100) UNIQUE,
    payment_status ENUM('pending','paid','failed') DEFAULT 'pending',
    payment_date TIMESTAMP NULL,
    FOREIGN KEY (registration_id) REFERENCES registrations(registration_id) ON DELETE CASCADE
);

CREATE TABLE entry_passes (
    pass_id INT AUTO_INCREMENT PRIMARY KEY,
    registration_id INT NOT NULL,
    pass_code VARCHAR(50) UNIQUE NOT NULL,
    issued_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at DATETIME,
    is_used BOOLEAN DEFAULT FALSE,
    used_at DATETIME,
    FOREIGN KEY (registration_id) REFERENCES registrations(registration_id) ON DELETE CASCADE
);

CREATE INDEX idx_students_college ON students(college_id);
CREATE INDEX idx_events_college ON events(college_id);
CREATE INDEX idx_events_category ON events(category_id);
CREATE INDEX idx_team_event ON teams(event_id);
CREATE INDEX idx_registration_event ON registrations(event_id);
CREATE INDEX idx_registration_student ON registrations(student_id);
CREATE INDEX idx_registration_team ON registrations(team_id);
CREATE INDEX idx_payment_registration ON payments(registration_id);
CREATE INDEX idx_pass_registration ON entry_passes(registration_id);
